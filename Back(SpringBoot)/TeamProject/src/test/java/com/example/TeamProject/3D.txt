import java.util.ArrayList;
import java.util.List;

public class PointCloudCenter {
    public static void main(String[] args) {
        // 3D 좌표를 나타내는 리스트 생성
        List<Point3D> coordinates = new ArrayList<>();
        coordinates.add(new Point3D(1.0, 2.0, 3.0));
        coordinates.add(new Point3D(4.0, 5.0, 6.0));
        coordinates.add(new Point3D(7.0, 8.0, 9.0));

        // 좌표의 개수
        int numCoordinates = coordinates.size();

        // X, Y, Z 좌표의 합계를 초기화
        double sumX = 0.0;
        double sumY = 0.0;
        double sumZ = 0.0;

        // 모든 좌표의 X, Y, Z 좌표를 합산
        for (Point3D coord : coordinates) {
            sumX += coord.getX();
            sumY += coord.getY();
            sumZ += coord.getZ();
        }

        // X, Y, Z 좌표의 평균을 계산하여 중심점 찾기
        double centerX = sumX / numCoordinates;
        double centerY = sumY / numCoordinates;
        double centerZ = sumZ / numCoordinates;

        // 중심점 출력
        System.out.println("중심점 좌표: X=" + centerX + ", Y=" + centerY + ", Z=" + centerZ);
    }
}

class Point3D {
    private double x;
    private double y;
    private double z;

    public Point3D(double x, double y, double z) {
        this.x = x;
        this.y = y;
        this.z = z;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public double getZ() {
        return z;
    }
}
